快速开始

Java 内嵌 (JimLangShell)

String script = """
function two() { return 2 ; } ;
var x = two() ;
println(x) ;
""";
JimLangShell shell = new JimLangShell();
shell.eval(script, null);

JSR-223 ScriptEngine

ScriptEngineManager manager = new ScriptEngineManager();
ScriptEngine engine = manager.getEngineByName("jim");
engine.eval("println(\"hello from JimLang\")");

语法速览

Control Flow (break/continue)

for (var i = 0; i < 5; i = i + 1) {
  if (i == 2) { continue }
  if (i == 4) { break }
  println(i)
}

Arrays & Objects

var arr = [1, 2, 3]
println(arr[0])     # 1
arr[1] = 10
println(arr.length) # 3

var person = { name: "Jim", age: 25 }
println(person.name)
person.age = 26
println(person.age)

示例

Fibonacci

function fibonacci(n) {
  var a = 0
  var b = 1
  for (var i = 0; i < n; i = i + 1) {
    var t = a
    a = b
    b = t + b
  }
  return a
}
println(fibonacci(10))

变量与函数

function add(a, b) { return a + b }
var x = 10
var y = 20
println(add(x, y))

文件 I/O

var path = "tmp.txt"
file_write(path, "hello")
println(file_exists(path))
println(file_read(path))
